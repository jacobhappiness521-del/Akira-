import { useState, useEffect } from 'react'
import ApiKeySetup from './ApiKeySetup.jsx'
import Chatbot from './Chatbot.jsx'

export default function App() {
  const [apiKey, setApiKey] = useState(null)

  // Check if a key exists in session storage on load
  useEffect(() => {
    const stored = sessionStorage.getItem('akira_api_key')
    if (stored) setApiKey(stored)
  }, [])

  const handleKeySubmit = (key) => {
    setApiKey(key)
  }

  const handleLogout = () => {
    sessionStorage.removeItem('akira_api_key')
    setApiKey(null)
  }

  if (!apiKey) {
    return <ApiKeySetup onKeySubmit={handleKeySubmit} />
  }

  return <Chatbot apiKey={apiKey} onLogout={handleLogout} />
}
